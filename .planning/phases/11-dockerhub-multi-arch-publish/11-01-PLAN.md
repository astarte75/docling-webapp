Create /Users/ricky/MyProjects/docling-webapp/Makefile with the following content exactly:

Variables:
- REGISTRY := docker.io
- NAMESPACE := ricky75
- TAG ?= latest  (note: ?= so it can be overridden at call time: make push TAG=v1.3)
- PLATFORMS := linux/amd64,linux/arm64
- BACKEND_IMAGE := $(REGISTRY)/$(NAMESPACE)/docling-webapp-backend:$(TAG)
- FRONTEND_IMAGE := $(REGISTRY)/$(NAMESPACE)/docling-webapp-frontend:$(TAG)
- BACKEND_LATEST := $(REGISTRY)/$(NAMESPACE)/docling-webapp-backend:latest
- FRONTEND_LATEST := $(REGISTRY)/$(NAMESPACE)/docling-webapp-frontend:latest

.PHONY targets: help builder push-backend push-frontend push

`help` target (default): prints available targets and TAG override example.

`builder` target:
  - docker buildx create --name multiarch --driver docker-container --bootstrap --use 2>/dev/null || docker buildx use multiarch
  - This is idempotent: creates builder on first run, reuses it on subsequent runs (Pitfall 5 from research)

`push-backend` target (depends on builder):
  - docker buildx build with --platform $(PLATFORMS)
  - Two --tag flags: $(BACKEND_IMAGE) and $(BACKEND_LATEST)
  - When TAG=latest both resolve to the same tag (harmless duplicate)
  - --file Dockerfile
  - --push
  - Context: . (project root)

`push-frontend` target (depends on builder):
  - Same pattern as push-backend but for Dockerfile.frontend
  - Two --tag flags: $(FRONTEND_IMAGE) and $(FRONTEND_LATEST)

`push` target (depends on push-backend push-frontend):
  - @echo summary of published images

Important implementation notes:
- Use tabs (not spaces) for Makefile recipe lines
- The double-tag approach (--tag $(IMAGE):$(TAG) --tag $(IMAGE):latest) ensures both versioned and latest tags are pushed in a single buildx command — this is the correct pattern per research
- Do NOT add --cache-from/--cache-to in this first version (cache not useful on first publish, adds complexity)
- Add a comment at the top: "# Requires: docker login && docker buildx (bundled with Docker Desktop)"
