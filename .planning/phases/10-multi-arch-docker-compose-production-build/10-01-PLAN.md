Create `Dockerfile.frontend` at project root (same level as existing `Dockerfile`):

```dockerfile
# Stage 1: Build React/Vite app
FROM node:22-alpine AS builder
WORKDIR /app
# Copy package files first for layer caching — npm ci only re-runs if package*.json changes
COPY frontend/package*.json ./
RUN npm ci --prefer-offline
COPY frontend/ .
RUN npm run build

# Stage 2: Serve with nginx
FROM nginx:1.27-alpine
COPY --from=builder /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/nginx.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

Create `nginx.conf` at project root:

```nginx
events {
    worker_connections 1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;
    sendfile      on;
    keepalive_timeout 65;

    server {
        listen 80;
        server_name _;
        root /usr/share/nginx/html;

        # React SPA fallback: React Router routes like /app/... return index.html
        location / {
            try_files $uri $uri/ /index.html;
        }

        # API reverse proxy — strips /api/ prefix, forwards to backend
        # proxy_buffering off covers SSE (/jobs/{id}/stream) — no separate location block needed
        location /api/ {
            proxy_pass         http://backend:8000/;
            proxy_http_version 1.1;
            proxy_set_header   Host $host;
            proxy_set_header   X-Real-IP $remote_addr;
            proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header   Connection '';
            proxy_buffering    off;
            proxy_read_timeout 86400s;
        }
    }
}
```

Create `.dockerignore` at project root (if it doesn't exist):

```
frontend/node_modules
frontend/dist
backend/__pycache__
backend/*.pyc
.git
.planning
```

Key constraints:
- `include /etc/nginx/mime.types;` is MANDATORY — without it, JS/CSS are served as text/plain and browser refuses to execute them
- `proxy_buffering off` on the entire `/api/` block — simpler than a separate SSE location block, no ordering issues
- `proxy_pass http://backend:8000/;` with trailing slash strips `/api/` prefix correctly — backend FastAPI routes start at `/convert`, `/jobs/`, etc. (no /api/ prefix)
- `npm ci` (not `npm install`) for reproducible builds
- `COPY package*.json ./` BEFORE `COPY frontend/ .` — Docker layer cache; npm ci only re-runs when package files change

Replace `docker-compose.yml` entirely with the production configuration:

```yaml
# Production Docker Compose — runs the full stack with a single command:
#   docker compose up --build -d
#
# For local development, use:
#   docker compose -f docker-compose.yml -f docker-compose.dev.yml up
#
# Environment variables (all optional — defaults cover standard deployments):
#   PORT              Host port exposed by nginx (default: 3000)
#   MAX_UPLOAD_SIZE_MB  Upload file size limit in MB (default: 50)

services:
  backend:
    build:
      context: .
      dockerfile: Dockerfile
    environment:
      - MAX_UPLOAD_SIZE_MB=${MAX_UPLOAD_SIZE_MB:-50}
    restart: unless-stopped
    # Backend not exposed directly — nginx talks to it via Docker network
    healthcheck:
      test: ["CMD", "python", "-c",
             "import urllib.request; urllib.request.urlopen('http://localhost:8000/health')"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 90s  # Docling model loading takes 30-60s on first start

  frontend:
    build:
      context: .
      dockerfile: Dockerfile.frontend
    ports:
      - "${PORT:-3000}:80"
    depends_on:
      backend:
        condition: service_healthy
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "wget", "-qO-", "http://localhost:80/"]
      interval: 10s
      timeout: 5s
      retries: 3
      start_period: 10s
```

Create `docker-compose.dev.yml` as development override:

```yaml
# Development override — use together with docker-compose.yml:
#   docker compose -f docker-compose.yml -f docker-compose.dev.yml up
#
# This override:
#   - Enables --reload for backend hot-reload
#   - Mounts backend source code as volume (no rebuild on code change)
#   - Exposes backend port 8000 directly for API testing

services:
  backend:
    command: uvicorn main:app --host 0.0.0.0 --port 8000 --reload
    volumes:
      - ./backend:/app
    ports:
      - "8000:8000"
    # Override healthcheck for faster dev iteration
    healthcheck:
      start_period: 30s
```

Key constraints:
- NO `--reload` in production docker-compose.yml
- NO volume mounts for code in production docker-compose.yml
- `start_period: 90s` on backend healthcheck — Docling loads large ML models at startup, 60s is the minimum but 90s is safer
- TESSDATA_PREFIX removed from docker-compose.yml environment — it's already set in Dockerfile ENV, no need to duplicate in compose
- `depends_on: condition: service_healthy` ensures nginx starts proxying only after backend passes health check
